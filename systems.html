local ReplicatedStorage = game:GetService("ReplicatedStorage")
local TweenService = game:GetService("TweenService")

local Remotes = ReplicatedStorage:WaitForChild("RemoteEvents")
local AnnoucmentRemote = Remotes:WaitForChild("Annoucment")
local Templates = ReplicatedStorage:WaitForChild("templates")
local AnnoucmentTemplate = Templates:WaitForChild("AnnoucmentTemplate")

local Container = script.Parent

local SlotHeight = AnnoucmentTemplate.Size.Y.Offset + 50
local DisplayTime = 4
local OpenTweenInfo = TweenInfo.new(0.5, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
local SlideTweenInfo = TweenInfo.new(0.35, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)

local Active = {}

local Eps = 0.001

local function tweenGradientOpen(gradient, tweenInfo)
	tweenInfo = tweenInfo or OpenTweenInfo

	local startLeft, startRight = 0.499, 0.501
	local progress = Instance.new("NumberValue")
	progress.Value = 0
	
	TweenService:Create(gradient.Parent:FindFirstChild("TextLabel"), SlideTweenInfo, {
		TextTransparency = 0,
		TextStrokeTransparency = 0
	}):Play()
	

	local function update()
		local alpha = progress.Value
		local leftPos = math.clamp(startLeft - (alpha * startLeft), Eps, 0.5 - Eps)
		local rightPos = math.clamp(startRight + (alpha * (1 - startRight)), 0.5 + Eps, 1 - Eps)

		gradient.Transparency = NumberSequence.new({
			NumberSequenceKeypoint.new(0, 1),
			NumberSequenceKeypoint.new(leftPos, 1),
			NumberSequenceKeypoint.new(0.5, 0),
			NumberSequenceKeypoint.new(rightPos, 1),
			NumberSequenceKeypoint.new(1, 1),
		})
	end

	update()
	progress.Changed:Connect(update)

	local tween = TweenService:Create(progress, tweenInfo, { Value = 1 })
	tween.Completed:Connect(function()
		progress:Destroy()
	end)
	tween:Play()

	return tween
end

local function reposition(animate)
	for index, frame in ipairs(Active) do
		local targetY = (index - 1) * SlotHeight
		local targetPos = UDim2.new(frame.Position.X.Scale, frame.Position.X.Offset, 0, targetY)

		if animate then
			TweenService:Create(frame, SlideTweenInfo, { Position = targetPos }):Play()
		else
			frame.Position = targetPos
		end
	end
end

local function dismiss(frame)
	local index = table.find(Active, frame)
	if not index then
		return
	end
	table.remove(Active, index)
	reposition(true)
	
	TweenService:Create(frame:FindFirstChild("TextLabel"), SlideTweenInfo, {
		TextTransparency = 1,
		TextStrokeTransparency = 1
	}):Play()
	local fadeOut = TweenService:Create(frame, SlideTweenInfo, {
		Position = frame.Position - UDim2.new(0, 0, 0, 12),
		BackgroundTransparency = 1,
	})
	fadeOut.Completed:Connect(function()
		frame:Destroy()
	end)
	fadeOut:Play()
end

local function createAnnouncement(message)
	local frame = AnnoucmentTemplate:Clone()
	frame.Position = UDim2.new(0.35, 0, 0, -SlotHeight)
	frame.Visible = true
	frame.Parent = Container

	table.insert(Active, 1, frame)
	reposition(true)

	local gradient = frame:FindFirstChildOfClass("UIGradient")
	if gradient then
		tweenGradientOpen(gradient)
	end

	task.delay(DisplayTime, function()
		dismiss(frame)
	end)
end

AnnoucmentRemote.OnClientEvent:Connect(function(message)
	createAnnouncement(message)
end)

local function onHit(weapon, victim)
    if victim:GetAttribute("Invincible") then
        return
    end
    local damage = weapon:GetAttribute("Damage") or 10
    victim.Humanoid:TakeDamage(damage)
end
