[Matchmaking system]

A matchmaking system with party support. Players queue together and are guaranteed to land in the same arena. Automatically promotes a new leader if the current one disconnects. Built across two decoupled modules — PartyService handles party state, ArenaService manages place assignment


-- PartyService: party creation & automatic leader reassignment
function PartyService:CreateParty(player)
	local Party = {
		Id = player.UserId,
		Leader = player,
		Members = {player},
		State = "Idle",
		Place = nil
	}
	UpdateParty:FireClient(player, Party)
	self.Parties[player.UserId] = Party
	return Party
end

function PartyService:RemoveMember(player)
	local party = self:GetParty(player)
	if not party then return end
	for i, member in ipairs(party.Members) do
		if member == player then
			table.remove(party.Members, i)
			break
		end
	end
	if party.Leader == player then
		party.Leader = party.Members[1] -- promote next member if leader leaves
	end
end

-- ArenaService: smart place finder, fills half-empty servers first
function ArenaService:FindPlaceForParty()
	local emptyPlace = nil
	for _, place in ipairs(placesFolder:GetChildren()) do
		local parties = place:GetAttribute("Parties") or 0
		if parties == 1 then
			return place -- prioritise filling existing servers
		end
		if parties == 0 and not emptyPlace then
			emptyPlace = place
		end
	end
	return emptyPlace
end

Notification System

A stacked announcement system that queues multiple notifications and reorganises them smoothly when one appears or disappears. Uses a NumberValue-driven tween to animate a UIGradient wipe-reveal since TweenService can't tween NumberSequence directly. Text fades in and out in sync with the panel animation.


-- NumberValue trick: TweenService can't tween NumberSequence directly
-- so we drive a hidden value and rebuild the gradient every change
local function tweenGradientOpen(gradient, tweenInfo)
	local startLeft, startRight = 0.499, 0.501
	local progress = Instance.new("NumberValue")
	progress.Value = 0

	local function update()
		local alpha = progress.Value
		local leftPos = math.clamp(startLeft - (alpha * startLeft), Eps, 0.5 - Eps)
		local rightPos = math.clamp(startRight + (alpha * (1 - startRight)), 0.5 + Eps, 1 - Eps)
		gradient.Transparency = NumberSequence.new({
			NumberSequenceKeypoint.new(0, 1),
			NumberSequenceKeypoint.new(leftPos, 1),
			NumberSequenceKeypoint.new(0.5, 0),
			NumberSequenceKeypoint.new(rightPos, 1),
			NumberSequenceKeypoint.new(1, 1),
		})
	end

	progress.Changed:Connect(update)
	local tween = TweenService:Create(progress, tweenInfo, { Value = 1 })
	tween.Completed:Connect(function() progress:Destroy() end)
	tween:Play()
end

-- stack manager: slide everything into place when the order changes
local function reposition(animate)
	for index, frame in ipairs(Active) do
		local targetPos = UDim2.new(frame.Position.X.Scale, frame.Position.X.Offset, 0, (index - 1) * SlotHeight)
		if animate then
			TweenService:Create(frame, SlideTweenInfo, { Position = targetPos }):Play()
		else
			frame.Position = targetPos
		end
	end
end

Settings menu

A client-side settings system supporting two control types — toggles and draggable sliders. Settings are applied instantly via player attributes, which any script in the game can read without needing remote events. The system is data-driven: new settings are added just by placing a UI element with the right attribute, no code changes needed.


-- toggle: switches between on/off states, updates player attribute instantly
local function setUpOffOn(option, frame, knob)
	local isOn = false
	local function update()
		knob.Position = isOn and UDim2.new(0.82, 0, 0, 0) or UDim2.new(0, 0, 0, 0)
		frame.BackgroundColor3 = isOn and Color3.fromRGB(0, 77, 0) or Color3.fromRGB(0, 0, 0)
		player:SetAttribute(option.Name, isOn)
	end
	frame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			isOn = not isOn
			update()
		end
	end)
	update()
end

-- slider: calculates value from mouse position relative to the frame
local function setUpDragable(option, frame, knob, volumetext, back)
	local dragging = false
	local function updateFromMouse()
		local relativeX = math.clamp(UIS:GetMouseLocation().X - frame.AbsolutePosition.X, 0, frame.AbsoluteSize.X)
		local percent = relativeX / frame.AbsoluteSize.X
		knob.Position = UDim2.new(percent, -knob.AbsoluteSize.X/2, 0.5, -knob.AbsoluteSize.Y/2)
		back.Size = UDim2.fromScale(percent, 1)
		volumetext.Text = math.floor(percent * 100) .. "%"
		player:SetAttribute(option.Name, 0.5 * (percent))
	end
	knob.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = true end end)
	UIS.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end)
	UIS.InputChanged:Connect(function(i)
		if dragging and i.UserInputType == Enum.UserInputType.MouseMovement then updateFromMouse() end
	end)
end
